228 static mat4 from_array(const float* arr) {
229 mat4 result;
230 for (int i = 0; i < 4; i++)
231 for (int j = 0; j < 4; j++)
232 result.columns[i][j] = arr[i * 4 + j];

236 inline mat4 operator*(const mat4& other) const {
237 mat4 result = zero();
238 for (int col = 0; col < 4; ++col) {
239 for (int row = 0; row < 4; ++row) {
240 float sum = 0.0f;
241 for (int k = 0; k < 4; ++k) {
242 sum += columns[k][row] * other.columns[col][k];
243 }
244 result.columns[col][row] = sum;
245 }
246 }
247 return result;
